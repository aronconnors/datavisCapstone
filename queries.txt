ARRIVALS:

SELECT
    destination_station_complex_name,
    month,
    day_of_week,
    hour_of_day,
    sum(estimated_average_ridership)
GROUP BY
    destination_station_complex_name,
    month,
    day_of_week,
    hour_of_day
ORDER BY
    destination_station_complex_name,
    month,
    day_of_week,
    hour_of_day


DEPARTURES:

SELECT
    origin_station_complex_name,
    month,
    day_of_week,
    hour_of_day,
    sum(estimated_average_ridership)
GROUP BY
    origin_station_complex_name,
    month,
    day_of_week,
    hour_of_day
ORDER BY
    origin_station_complex_name,
    month,
    day_of_week,
    hour_of_day
